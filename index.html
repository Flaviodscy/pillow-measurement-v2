        class MeasurementApp {

            constructor() {

                this.video = document.getElementById('camera-feed');

                this.canvas = document.getElementById('ar-overlay');

                this.ctx = this.canvas.getContext('2d');

                this.measurements = [];

                this.currentMeasurement = null;

                this.unit = 'cm';

                this.pixelsPerCm = 10;

                

                // Detect environment

                this.isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) && !window.MSStream;

                this.isAndroid = /Android/.test(navigator.userAgent);

                this.isInIframe = window.self !== window.top;

                

                this.init();

            }


            async init() {

                // Show warning if in iframe (Claude preview)

                if (this.isInIframe) {

                    console.log('Running in iframe - camera access may be restricted');

                }

                

                document.getElementById('start-camera').addEventListener('click', () => this.startCamera());

                document.getElementById('add-point').addEventListener('click', () => this.addPoint());

                document.getElementById('clear-all').addEventListener('click', () => this.clearAll());

                document.getElementById('unit-toggle').addEventListener('click', () => this.toggleUnit());

                

                window.addEventListener('resize', () => this.resizeCanvas());

                

                // Prevent default touch behaviors

                document.addEventListener('touchmove', (e) => {

                    if (e.target.closest('.measurement-point')) {

                        e.preventDefault();

                    }

                }, { passive: false });

            }


            async startCamera() {

                try {

                    if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {

                        throw new Error('Camera API not supported');

                    }


                    // Simple constraints that work across all devices

                    const constraints = {

                        audio: false,

                        video: {

                            facingMode: 'environment',

                            width: { ideal: 1920 },

                            height: { ideal: 1080 }

                        }

                    };


                    // Get camera stream

                    const stream = await navigator.mediaDevices.getUserMedia(constraints);

                    

                    // Set up video element

                    this.video.srcObject = stream;

                    

                    // Ensure video plays

                    await this.video.play();

                    

                    // Hide permission overlay once video is playing

                    this.video.addEventListener('playing', () => {

                        document.getElementById('permission-overlay').style.display = 'none';

                        this.resizeCanvas();

                        this.animate();

                    }, { once: true });


                } catch (error) {

                    console.error('Camera error:', error);

                    

                    let errorMessage = '';

                    

                    if (this.isInIframe) {

                        errorMessage = 'Camera access is blocked in this preview. Please follow the instructions above to use the app.';

                    } else if (error.name === 'NotAllowedError') {

                        errorMessage = 'Camera permission denied. Please allow camera access in your browser settings.';

                    } else if (error.name === 'NotFoundError') {

                        errorMessage = 'No camera found on this device.';

                    } else {

                        errorMessage = `Camera error: ${error.message}`;

                    }

                    

                    document.getElementById('error-message').textContent = errorMessage;

                }

            }


            resizeCanvas() {

                this.canvas.width = window.innerWidth;

                this.canvas.height = window.innerHeight;

            }


            addPoint() {

                const centerX = window.innerWidth / 2;

                const centerY = window.innerHeight / 2;

                

                if (!this.currentMeasurement) {

                    this.currentMeasurement = {

                        start: { x: centerX, y: centerY },

                        end: null,

                        id: Date.now()

                    };

                    this.createPointElement(centerX, centerY, 'start', this.currentMeasurement.id);

                } else if (!this.currentMeasurement.end) {

                    this.currentMeasurement.end = { x: centerX, y: centerY };

                    this.createPointElement(centerX, centerY, 'end', this.currentMeasurement.id);

                    this.measurements.push(this.currentMeasurement);

                    this.createMeasurementLine(this.currentMeasurement);

                    this.currentMeasurement = null;

                    this.updateMeasurementCount();

                }

            }


            createPointElement(x, y, type, measurementId) {

                const point = document.createElement('div');

                point.className = 'measurement-point';

                point.dataset.type = type;

                point.dataset.measurementId = measurementId;

                point.style.left = x + 'px';

                point.style.top = y + 'px';

                point.innerHTML = '<div class="point-inner"></div>';

                

                // Handle both touch and mouse events

                let isDragging = false;

                let startX, startY;

                

                const handleStart = (e) => {

                    isDragging = true;

                    const touch = e.touches ? e.touches[0] : e;

                    startX = touch.clientX - x;

                    startY = touch.clientY - y;

                    e.preventDefault();

                };

                

                const handleMove = (e) => {

                    if (!isDragging) return;

                    const touch = e.touches ? e.touches[0] : e;

                    const newX = touch.clientX - startX;

                    const newY = touch.clientY - startY;

                    

                    point.style.left = newX + 'px';

                    point.style.top = newY + 'px';

                    

                    this.updateMeasurement(measurementId, type, newX, newY);

                    e.preventDefault();

                };

                

                const handleEnd = () => {

                    isDragging = false;

                };

                

                // Touch events

                point.addEventListener('touchstart', handleStart, { passive: false });

                point.addEventListener('touchmove', handleMove, { passive: false });

                point.addEventListener('touchend', handleEnd);

                

                // Mouse events as fallback

                point.addEventListener('mousedown', handleStart);

                point.addEventListener('mousemove', handleMove);

                point.addEventListener('mouseup', handleEnd);

                

                document.getElementById('app').appendChild(point);

            }


            createMeasurementLine(measurement) {

                const line = document.createElement('div');

                line.className = 'measurement-line';

                line.dataset.measurementId = measurement.id;

                

                const label = document.createElement('div');

                label.className = 'measurement-label';

                label.dataset.measurementId = measurement.id;

                

                document.getElementById('app').appendChild(line);

                document.getElementById('app').appendChild(label);

                

                this.updateLine(measurement);

            }


            updateMeasurement(id, type, x, y) {

                const measurement = this.measurements.find(m => m.id === id);

                if (measurement) {

                    measurement[type] = { x, y };

                    this.updateLine(measurement);

                }

            }


            updateLine(measurement) {

                const line = document.querySelector(`.measurement-line[data-measurement-id="${measurement.id}"]`);

                const label = document.querySelector(`.measurement-label[data-measurement-id="${measurement.id}"]`);

                

                if (!line || !label) return;

                

                const dx = measurement.end.x - measurement.start.x;

                const dy = measurement.end.y - measurement.start.y;

                const distance = Math.sqrt(dx * dx + dy * dy);

                const angle = Math.atan2(dy, dx) * 180 / Math.PI;

                

                line.style.left = measurement.start.x + 'px';

                line.style.top = measurement.start.y + 'px';

                line.style.width = distance + 'px';

                line.style.transform = `rotate(${angle}deg)`;

                

                const midX = (measurement.start.x + measurement.end.x) / 2;

                const midY = (measurement.start.y + measurement.end.y) / 2;

                

                const measurementValue = this.calculateDistance(distance);

                label.textContent = measurementValue;

                label.style.left = midX + 'px';

                label.style.top = (midY - 30) + 'px';

                label.style.transform = 'translate(-50%, -50%)';

            }


            calculateDistance(pixels) {

                const value = pixels / this.pixelsPerCm;

                if (this.unit === 'cm') {

                    return value.toFixed(1) + ' cm';

                } else if (this.unit === 'm') {

                    return (value / 100).toFixed(2) + ' m';

                } else {

                    return (value / 2.54).toFixed(1) + ' in';

                }

            }


            toggleUnit() {

                const units = ['cm', 'm', 'in'];

                const currentIndex = units.indexOf(this.unit);

                this.unit = units[(currentIndex + 1) % units.length];

                document.getElementById('unit-toggle').textContent = this.unit;

                

                this.measurements.forEach(m => this.updateLine(m));

            }


            clearAll() {
